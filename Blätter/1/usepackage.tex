\newcommand{\makeanswerbox}[1]{
	\def\x{#1}
	\noindent
	\begin{tikzpicture}
		\draw[blur shadow={shadow blur steps=10},fill = white] (0,0) rectangle (\textwidth,\x);
	\end{tikzpicture}
}

\newcommand{\makecustomanswerbox}[2]{
	\def\x{#1}
	\def\y{#2}
	\noindent
	\begin{tikzpicture}
	\draw (0,0) rectangle (\x,\y);
	\end{tikzpicture}
}

{

}
